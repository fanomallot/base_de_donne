CREATE TABLE 'user'(
'id' INTEGER PRIMARY KEY AUTOICREMENT,
'name' TEXT);
CREATE TABLE 'article'('id' INTEGER PRIMARY KEY AUTOINCREMENT,'user_id' INTEGER ,'cat_id' INTEGER , 'titre' TEXT ,FOREIGN KEY (user_id) REFERENCES user (id),FOREIGN KEY (cat_id) REFERENCES categorie (id) );
CREATE TABLE 'categorie'('id' INTEGER PRIMARY KEY AUTOINCREMENT,'tag_id' INTEGER ,'titre' TEXT ,FOREIGN KEY (tag_id) REFERENCES tag (id) );
CREATE TABLE 'tag'('id' INTEGER PRIMARY KEY AUTOINCREMENT, 'titre' TEXT , 'color' TEXT );
